@int arr[6] = {64, 34, 25, 12, 22, 11} $
@int n is 6 $
@int i $
@int j $
@int key $

iterate (i = 1 $ i < n $ i = i + 1) {
    key = arr[i] $
    j = i - 1 $
    
    loop (j >= 0 and arr[j] > key) {
        arr[j + 1] = arr[j] $
        j = j - 1 $
    }
    arr[j + 1] = key $
}

iterate (i = 0 $ i < n $ i = i + 1) {
    show(arr[i]) $
}
