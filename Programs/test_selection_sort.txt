@int arr[6] = {64, 25, 12, 22, 11, 34} $
@int n is 6 $
@int i $
@int j $
@int min_idx $
@int temp $

iterate (i = 0 $ i < n - 1 $ i = i + 1) {
    min_idx = i $
    
    iterate (j = i + 1 $ j < n $ j = j + 1) {
        if (arr[j] < arr[min_idx]) {
            min_idx = j $
        }
    }
    
    if (min_idx != i) {
        temp = arr[i] $
        arr[i] = arr[min_idx] $
        arr[min_idx] = temp $
    }
}

iterate (i = 0 $ i < n $ i = i + 1) {
    show(arr[i]) $
}
